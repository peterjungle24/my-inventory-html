Step 4 --> Scripts for use

This is where I will define the scripts

Centralize (centra.cs)

Will handle the [auto-center] button function

put this script in Button object

                    
using UnityEngine;
namespace Inventory
{
    /// <summary> 
    /// Uses this class to centralize the item for the slot. 
    /// </summary>
    public class centra : MonoBehaviour
    {
        void Update()
        {
            //put that on Button
            //this code will centralize the buttons to the center of the slot.
            transform.position += (transform.parent.position - transform.position) * 4 * Time.deltaTime;
        }
    }
}

Inventory Methods (inv.cs)

Will handle somethings about the inventory

put this script in Canvas object

                    
using UnityEngine;

public class inv : MonoBehaviour
{
    public GameObject mouseItem;    //object that will be centralized.
    public item[] item;             //item

    public void DragItem(GameObject btn)
    {
        mouseItem = btn;
        mouseItem.transform.position = Input.mousePosition;
    }

    public void DropItem(GameObject btn)
    {
        if(mouseItem != null)   //if its using the Mouse
        {
            Transform aux = mouseItem.transform.parent;

            mouseItem.transform.SetParent(btn.transform.parent);
            btn.transform.SetParent(aux);
        }
    }
}
